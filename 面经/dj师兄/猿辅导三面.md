6. 从左上到右下 45° 打印二维数组

```javascript
// 左上到右下
function printDiagonal(arr) {
  const rows = arr.length;
  const cols = arr[0].length;
  let row = 0;
  let col = 0;
  while (row < rows && col < cols) {
    console.log(arr[row][col]);
    row++;
    col++;
    if (row >= rows || col >= cols) break;
  }
}
```
